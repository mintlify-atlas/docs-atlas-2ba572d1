## Overview

CV Builder uses a strongly-typed data structure defined in `lib/types.ts`. All resume data is stored as a single `CVData` object containing personal information, experience, education, and other sections.

## Core Data Type

The main data structure that represents a complete resume:

```typescript lib/types.ts
export interface CVData {
  personalInfo: PersonalInfo;
  experience: Experience[];
  education: Education[];
  projects: Project[];
  achievements: Achievement[];
  languages: Language[];
  skills: Skill[];
  sectionOrder: SectionId[];
  references: string;
  hiddenSections: SectionId[];
  template?: TemplateId;
}
```

<ParamField path="personalInfo" type="PersonalInfo" required>
  User's personal contact information and profile
</ParamField>

<ParamField path="experience" type="Experience[]" required>
  Array of work experience entries
</ParamField>

<ParamField path="education" type="Education[]" required>
  Array of education entries
</ParamField>

<ParamField path="projects" type="Project[]" required>
  Array of personal or professional projects
</ParamField>

<ParamField path="achievements" type="Achievement[]" required>
  Array of awards, certifications, or achievements
</ParamField>

<ParamField path="languages" type="Language[]" required>
  Array of languages with proficiency levels
</ParamField>

<ParamField path="skills" type="Skill[]" required>
  Array of technical and professional skills
</ParamField>

<ParamField path="sectionOrder" type="SectionId[]" required>
  Order in which sections appear in the resume
</ParamField>

<ParamField path="references" type="string" required>
  References text (e.g., "Available upon request")
</ParamField>

<ParamField path="hiddenSections" type="SectionId[]" required>
  Sections hidden from preview and PDF export
</ParamField>

<ParamField path="template" type="TemplateId">
  Selected resume template ('default' | 'rhyhorn' | 'nexus')
</ParamField>

## Merge Types

Types used for conflict resolution during sync:

### FieldConflict

```typescript lib/types.ts
export interface FieldConflict {
  section: SectionId | 'sectionOrder' | 'hiddenSections' | 'template' | 'references';
  field?: string;  // For nested fields like personalInfo.fullName, or item ID for arrays
  localValue: unknown;
  serverValue: unknown;
  baseValue: unknown;
  description?: string; // Human-readable description of the conflict
}
```

<ResponseField name="section" type="string" required>
  Section where the conflict occurred
</ResponseField>

<ResponseField name="field" type="string">
  Specific field name or item ID
</ResponseField>

<ResponseField name="localValue" type="unknown" required>
  Value from local (browser) storage
</ResponseField>

<ResponseField name="serverValue" type="unknown" required>
  Value from Firebase server
</ResponseField>

<ResponseField name="baseValue" type="unknown" required>
  Value from last sync (common ancestor)
</ResponseField>

<ResponseField name="description" type="string">
  Human-readable conflict description
</ResponseField>

### MergeResult

```typescript lib/types.ts
export interface MergeResult {
  merged: CVData;
  conflicts: FieldConflict[];
  hasConflicts: boolean;
}
```

<ResponseField name="merged" type="CVData" required>
  Merged resume data
</ResponseField>

<ResponseField name="conflicts" type="FieldConflict[]" required>
  Array of detected conflicts
</ResponseField>

<ResponseField name="hasConflicts" type="boolean" required>
  Whether any conflicts were detected
</ResponseField>

## Usage Example

```typescript
import { CVData, initialCVData } from '@/lib/types';

// Create a new resume
const myResume: CVData = {
  ...initialCVData,
  personalInfo: {
    ...initialCVData.personalInfo,
    fullName: "John Doe",
    email: "john@example.com",
    jobTitle: "Software Engineer",
  },
  experience: [
    {
      id: "exp_1",
      company: "Tech Corp",
      role: "Senior Developer",
      startDate: "2020-01",
      endDate: "2024-03",
      current: true,
      description: "Led development of core features",
    },
  ],
};
```
